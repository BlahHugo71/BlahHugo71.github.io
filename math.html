Math Game (mostly a clone of Clicker)

A Clicker Game

a = 0

a = a + 1

a_per_sec = 0

every second, a = a + a_per_sec

a = a - 50, a_per_sec = a_per_sec + 1

a = a - 800, a_per_sec = a_per_sec + 12

a = a - 6400, a_per_sec = a_per_sec + 75

a = a - 70000, a_per_sec = a_per_sec + 800

a = a - 500000, a_per_sec = a_per_sec + 9600
